    Aplicativo client app = angular.module(“vanessador”, [“ngRoute”, “ngResource”, “ngCookies”, “toastr”])

    window.app = app = null

Vue.http.get('/templates')
        .then (templates) ->
                routes = path: t.route, component: {template: t.template} for t in templates
                router = new VueRouter routes: routes
                app = new Vue(router: router).$mount("#app")
        .catch (e) -> console.log e
